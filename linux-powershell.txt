-- 新建文件夹
mkdir xxx 

-- 强制删除文件夹xxx ！慎用
rmdir xxx -Recurse -Force   # powershell
rm -rf xxx # linux 递归删除文件夹
rm -r xxx  # linux 普通删除文件夹

-- 新建文件
touch xxx.txt

-- 复制文件、文件夹、
cp a.txt b.txt  # 复制文件内容
cp -r xxx1 xxx2  # xxx1整个文件夹复制到xxx2(---> xxx2下的子文件夹xxx1)

-- 重命名文件/文件夹
mv a.txt b.txt
mv xxx1 xxx2

-- 查看文件
cat 文件名   # 全部输出
less 文件名  # 分页查看 
↑ ↓：滚动
/关键词：搜索
q：退出
head -n 20 文件名  # 前20行
tail -n 20 文件名  # 最后10行
tail -f 文件名     # 实时查看/会更新的文件，如log
grep "关键词" 文件名  # 查文件中指定内容

-- 修改文件 
vim 文件名
i     # 插入
:wq  # 保存退出
:q！ # 退出不保存
